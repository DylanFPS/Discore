Example - !Ping bot

using System;
using Discore;

namespace example_bot
{
    public class Program
    {
        const string TOKEN = "";

        public static void Main(string[] args)
        {
            DiscordClient myBot = new DiscordClient();

            myBot.OnConnected += (sender, e) => {
                Console.WriteLine($"Connected as {myBot.User}!");
            };

            myBot.OnMessageCreated += (sender, e) => {
                DiscordMessage message = e.Message;
                if (message.Author != myBot.User //if we didnt post it
		&& message.Content.Contains("!ping")) //and contains !ping
                {
                    message.Channel.SendMessage("pong!");
                }
            };

            if (!myBot.Connect(TOKEN).Result)
            {
                Console.WriteLine("Failed to connect!");
            }

        }
    }
}
